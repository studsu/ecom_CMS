These variables are automatically available in your templates through Django's context processors and views.

Global Variables

Variable Type Description Example
{{ SITE_NAME }} String Site name from settings "My Store"
{{ CURRENCY_SYMBOL }} String Currency symbol "‚Çπ", "$", "‚Ç¨"

{{ request }} Object Current HTTP request Django request object

Product Variables

Variable Type Description Usage
{{ product.title }} String Product name Display product title
{{ product.slug }} String URL-friendly identifier Generate product URLs
{{ product.price }} Decimal Original price Show original price
{{ product.sale_price }} Decimal Sale price (if on sale) Show discounted price
{{ product.is_on_sale }} Boolean Whether product is on sale Conditional sale display
{{ product.is_in_stock }} Boolean Stock availability Enable/disable add to cart
{{ product.image }} Image Main product image Display product image
{{ product.category }} Object Product category Show category info
{{ product.variants.all }} QuerySet Product variants Show size/color options

Cart Variables

Variable Type Description Usage
{{ cart_items }} QuerySet Items in cart Display cart contents
{{ cart_total }} Decimal Total cart value Show total price

JavaScript Functions

// Variant Quantity Management
function decreaseVariantQty(variantId) {
    const input = document.getElementById(`variant-qty-${variantId}`);
    if (input.value > 0) {
        input.value = parseInt(input.value) - 1;
    }
}

function increaseVariantQty(variantId) {
    const input = document.getElementById(`variant-qty-${variantId}`);
    const max = parseInt(input.getAttribute('max'));
    if (parseInt(input.value) < max) {
        input.value = parseInt(input.value) + 1;
    }
}

// Add Selected Variants to Cart
function addSelectedVariantsToCart() {
    const variants = [];

    // Collect all variant quantities
    document.querySelectorAll('[id^="variant-qty-"]').forEach(input => {
        const variantId = input.id.split('-')[2];
        const quantity = parseInt(input.value);

        if (quantity > 0) {
            variants.push({
                variant_id: variantId,
                quantity: quantity
            });
        }
    });

    if (variants.length === 0) {
        alert('Please select at least one variant');
        return;
    }

    // Submit to cart
    fetch('/cart/add-variants/', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'X-CSRFToken': getCookie('csrftoken')
        },
        body: JSON.stringify({ variants: variants })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            location.reload();
        } else {
            alert(data.message || 'Error adding to cart');
        }
    });
}

// Get CSRF Cookie
function getCookie(name) {
    let cookieValue = null;
    if (document.cookie && document.cookie !== '') {
        const cookies = document.cookie.split(';');
        for (let i = 0; i < cookies.length; i++) {
            const cookie = cookies[i].trim();
            if (cookie.substring(0, name.length + 1) === (name + '=')) {
                cookieValue = decodeURIComponent(cookie.substring(name.length + 1));
                break;
            }
        }
    }
    return cookieValue;
}

‚≠ê Best Practices

‚úÖ Do's

  • Use semantic HTML5 elements
  • Implement responsive design
  • Include proper alt text for images
  • Use CSRF tokens in forms
  • Follow Django template conventions
  • Test across different devices

‚ùå Don'ts

  • Don't hardcode URLs in templates
  • Don't forget to handle empty states
  • Don't ignore mobile responsiveness
  • Don't use inline styles excessively
  • Don't skip error handling
  • Don't forget to escape user content

Performance Tips

üöÄ Performance Optimization

  • Image Optimization: Use appropriate image formats and sizes
  • CSS Minification: Minimize CSS files in production
  • Lazy Loading: Implement lazy loading for images
  • Caching: Use Django's template caching where appropriate
  • CDN: Serve static files from CDN in production

Security Considerations

üîí Security Best Practices

  • Always use CSRF protection in forms
  • Sanitize user input display with Django's auto-escaping
  • Use HTTPS in production
  • Validate file uploads properly
  • Don't expose sensitive information in templates
